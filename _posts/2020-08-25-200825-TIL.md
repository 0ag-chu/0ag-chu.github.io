      def flatten_dict(d):
          l = []
          for k, v in sorted(d.items()):
              if isinstance(v, dict):
                  flatten_v = flatten_dict(v)
                  for my_l in reversed(flatten_v):
                      my_l.insert(0, k)
                  l.extend(flatten_v)
              elif isinstance(v, list):
                  for l_val in v:
                      l.append([k, l_val])
              else:
                  l.append([k, v])
          return l
